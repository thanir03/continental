const ListItem: React.FC<{ item: Listings; viewableItems: SharedValue<ViewToken[]> }> = React.memo(({ item, viewableItems }) => {
  const rStyle = useAnimatedStyle(() => {
    // Check if the item is visible
    const isVisible = viewableItems.value.some(vItem => vItem.isViewable && vItem.item.hotelId === item.hotelId);

    return {
      opacity: withTiming(isVisible ? 1 : 0),
      transform: [
        {
          scale: withTiming(isVisible ? 1 : 0.5),
        },
      ],
    };
  }, [viewableItems, item]);
